$ git status // seve para observar se os arquivos estarão no controle de versão do git

$ git add "nome_do_arquivo" // passa um arquivo isolado para o o controle de versão

$ git add . // adiciona todos os arquivos da pasta para versão de controle

$ git commit -m "commit inicial" // recordar as mudanças no repositório local e uma -m (mensagem)

$ git config --global user.email "meuemail@email.com" // essas duas partes são para configurar o git e conectá-lo com o github
$ git config --global user.name "Meu nome" // utilizando seu usuário

$ git remote add origin https://github.com/JulioRobocop/projeto1_entendendo_git // definir para onde o código sera enviado
$ git push --set-upstream origin master // enviar inicialmente para a master

// Para atualizar alguma coisa, deve-se checar os status primeiramente: 
$ git status

//Adicionar os arquivos modificados ou simples arquivo:
$ git add "nome_do_arquivo" // esse ou o de baixo
$ git add .

// mandar um commit com uma mensagem falando oq foi alterado
$ git commit -m "resumir a mudança"

// enviar para o repositório do github
$ git push

// verificar os histórico
$ git reflog

// mudar para as versões antigas localmente
$ git reset --hard XXXX // passar o código da atualização

//verificar as branchs do meu projeto
$ git branch

//criar uma branch
$ git branch "nome da branch"

// altera de branch no seu projeto
$ git checkout "nome da branch"

// Unir duas branch, na branch principal (tem que estar nessa) voce puxa o repositório da outra branch
$ git merge "o nome da outra branch de onde voce quer puxar as coisas e trazer para esta"

// Antes de realizar o merg, entre na sua branch principal e faça isso
$ git pull // para trazer as atualizações que foram feitas no site, para minha, evitando conflito entre as pessoas que estão fazendo o código

// e enfim, jogar de volta para o repositório no servidor
$ git push

// criar uma branch baseada na atual (NÃO USAR AS ASPAS)
$ git checkout -b "nome da branch" "nome da branch base"
